Scopes in HTML <px-app>, <px-scope>

HTML
<div px-app px-scope="main-scope">
    <input type="text" px-input="name">
    Welcome <span px-bind="name"></span>!
</div>
				
JavaScript
var myApp = pxApp(); // initializing application

myApp.scope('main-scope', function(px) {
	// function within scope
});

Two-way data binding px.set(name, value)

HTML
<div px-app px-scope="main-scope">
    <input type="text" px-input="name">
    Welcome <span px-bind="name"></span>!
</div>
				
JavaScript
var myApp = pxApp();

myApp.scope('main-scope', function(px) {
    px.set('name', 'Martin'); // Set name to 'Martin'
    px.get('name');           // Get name variable
});
Welcome !

Watching variables px.watch(name, callback)

HTML
<div px-app px-scope="main-scope">
    <input type="text" px-input="name">
    Welcome <span px-bind="name"></span>!<br>
    Changes: <span px-bind="count"></span>
</div>
				
JavaScript
var myApp = pxApp();

myApp.scope('main-scope', function(px) {
    px.set('name', 'Martin');
    px.set('count', 0)
    px.watch('name', function() {
    	px.set('count', px.get('count') + 1);
    });
});
